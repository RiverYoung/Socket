{

	strcat(dirPath,"/mnt/usb/");

		return false;
	}
	while(NULL != (dirconnt=readdir(dir_1)))
	{

			continue;

	   strcat(dirPath,dirconnt->d_name);

	   dir_2 = opendir(dirPath);

	   {
	   		continue;
	   }

	   {	    
	   	  if(strncmp(filePath,dirconnt_2->d_name,len) == 0)
	   	  {
	   	    closedir(dir_1);
		    closedir(dir_2);
	   	  	return true;
	   	  }
	   }
	   
	}

     closedir(dir_2);

}
